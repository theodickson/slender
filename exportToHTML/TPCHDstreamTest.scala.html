171      def genQuery[V,ID](rdds: TpchRdds, lineitem: LiteralExpr[V,ID]) = { 
172        val customer = Bag(rdds.customer, "customer".narrow) 
173        val orders = Bag(rdds.orders, "orders".narrow) 
174        val part = Bag(rdds.part, "part".narrow) 
175   
176        val (orderKey0, partName0) = Vars("X1", "X2") 
177        val (partNames0, custKey0, orderDate0) = Vars("Y1", "Y2", "Y3") 
178        val (orderDate1, partNames1, custName0) = Vars("Z1", "Z2", "Z3") 
179   
180        //Drop the suppkeys to get a Bag[(partkey,orderkey)] 
181        val partOrders = For((l_orderkey, l_partkey, __) <-- lineitem) Yield(l_partkey, l_orderkey) 
182   
183        //join with part to get a Bag[(partkey,(orderKey,partName))], drop the partkeys and group to get 
184        //a Bag[(orderKey,Bag[partName])] 
185        val orderPartNames = Group( 
186          For((__, orderKey0, partName0) <-- partOrders.join(part)) Yield(orderKey0, partName0) 
187        ) 
188        //join with orders and reshape to get a Bag[(custKey,orderDate,Bag[partName])] 
189        val customerOrders = 
190          For((__, partNames0, custKey0, orderDate0) <-- orderPartNames.join(orders)) Yield(custKey0, orderDate0, partNames0) 
191   
192        //join with customers and reshape to get a Bag[(custName,orderDate,Bag[partName])] 
193        val customerNameOrders = 
194          For((__, orderDate1, partNames1, custName0, __) <-- customerOrders.join(customer)) Yield 
195            (custName0, orderDate1, partNames1) 
196   
197        //finally group to get a Bag[(custName,Bag[(orderDate,Bag[partName])])] 
198        val query = Group(customerNameOrders) 
199        query 
200      }

202      def testGen(sampleName: String, 
203                  batchSize: Int, 
204                  shredded: Boolean, 
205                  acc: Boolean): ((SparkSession,StreamingContext) => Unit) = (spark: SparkSession, ssc: StreamingContext) => { 
206   
207        println(s"batchSize = $batchSize, shredded = $shredded, acc = $acc") 
208        val rdds = new TpchRdds(sampleName)(spark) 
209        val inputs = List(rdds.lineitem, rdds.part, rdds.orders, rdds.customer) 
210        val dstream = rddToDStream(rdds.lineitem, batchSize, ssc) 
211        val lineitem = Bag(dstream, "lineitem".narrow) 
212        val query = genQuery(rdds, lineitem) 
213        if (!shredded) query.eval.profile(batchSize,inputs,spark,ssc) 
214        else query.shreddedEval.profile(batchSize,inputs,acc,spark,ssc) 
215        println() 
216      } 
217      val tests = new collection.mutable.ArrayBuffer[DStreamTest] 
218      batchSizes.foreach { batchSize => 
219        tests += testGen(sampleName,batchSize,false,false)

221        tests += testGen(sampleName,batchSize,true,true) 
222      } 
223      println(s"Query - Q1. Dataset - $sampleName") 
224      tests.foreach(singleDStreamTest) 
225      println() 
226    }

230      /**Q2 with ONLY orders streamed*/ 
231      def genQuery[V,ID](rdds: TpchRdds, orders: LiteralExpr[V,ID]) = { 
232        val custNames = Var("X") 
233        val lineitem = Bag(rdds.lineitem, "lineitem".narrow) 
234        val customer = Bag(rdds.customer, "customer".narrow) 
235        val supplier = Bag(rdds.supplier, "supplier".narrow) 
236   
237        //First join orders with lineitems on orderkey to get all pairs of custkey/suppkey 
238        val custSuppliers = For ((__,o_custkey,__,__,l_suppkey) <-- orders.join(lineitem)) Yield (o_custkey,l_suppkey) 
239        //Then join customer with the above to get the customers names, yielding all pairs of suppkey/custname. 
240        //Then group this to get each supplier with its bag of customer names who have used it: 
241        val inner = For ((__,c_name,__,l_suppkey) <-- customer.join(custSuppliers)) Yield (l_suppkey,c_name) 
242        val supplierCustomers = Group( 
243          For ((__,c_name,__,l_suppkey) <-- customer.join(custSuppliers)) Yield (l_suppkey,c_name) 
244        ) 
245        //Finally join with supplier to get the suppliers names: 
246        val query = For ((__,s_name,__,custNames) <-- supplier.join(supplierCustomers)) Yield (s_name,custNames) 
247        query 
248      }

279      def genQuery[V,ID](rdds: TpchRdds, partSupp: LiteralExpr[V,ID]) = { 
280        /**For each part, return the name, the names and nations of all suppliers who supply it, 
281        * and the names and nations of all customers who've ordered it 
282        * */ 
283   
284        val (suppliers,customers) = Vars("SUPS","CUSTS") 
285        val lineitem = Bag(rdds.lineitem, "lineitem".narrow) 
286        val customer = Bag(rdds.customer, "customer".narrow) 
287        val supplier = Bag(rdds.supplier, "supplier".narrow) 
288        val part = Bag(rdds.part, "part".narrow) 
289        val orders = Bag(rdds.orders, "orders".narrow) 
290   
291        //First join partSupp to suppliers and group to get partkeys and their bags of supplier names 
292        val partSupp2 = For ((ps_partkey,ps_suppkey) <-- partSupp) Yield (ps_suppkey,ps_partkey) 
293        val partSuppliers = Group( 
294          For ((ps_suppkey,ps_partkey,s_name,s_nationkey) <-- partSupp2.join(supplier)) Yield (ps_partkey,(s_name,s_nationkey)) 
295        ) 
296   
297        val partCustomerPairs = For ((__,l_partkey,__,o_custkey,__) <-- lineitem.join(orders)) Yield (l_partkey,o_custkey) 
298        val partCustomers = Group( 
299          For ((l_partkey,__,c_name,c_nationkey) <-- partCustomerPairs.join(customer)) Yield (l_partkey,(c_name,c_nationkey)) 
300        ) 
301   
302        val query = 
303          For ( 
304            (__,p_name,suppliers,customers) <-- part.join(partSuppliers.join(partCustomers)) 
305          ) Yield (p_name,suppliers,customers) 
306   
307        query 
308      }
