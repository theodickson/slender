41     test("Group isolation test - lineitem") { 
42       def testGen(sampleName: String, 
43                   batchSize: Int, 
44                   shredded: Boolean, 
45                   acc: Boolean): ((SparkSession,StreamingContext) => Unit) = (spark: SparkSession, ssc: StreamingContext) => { 
46         println(s"batchSize = $batchSize, shredded = $shredded, acc = $acc") 
47         val rdds = new TpchRdds(sampleName)(spark) 
48         val dstream = rddToDStream(rdds.lineitem, batchSize, ssc) 
49         val lineitem = Bag(dstream, "lineitem".narrow) 
50         val query = Group( 
51           For ((l_orderkey,l_partkey,l_suppkey) <-- lineitem) Yield (l_orderkey,l_partkey) 
52         ) 
53         if (!shredded) query.eval.profile(batchSize,List(rdds.lineitem),spark,ssc) 
54         else query.shreddedEval.profile(batchSize,List(rdds.lineitem),acc,spark,ssc) 
55         println() 
56       } 
57       val tests = new collection.mutable.ArrayBuffer[DStreamTest] 
58       batchSizes.foreach { batchSize => 
59         tests += testGen(sampleName,batchSize,false,false) 
60         tests += testGen(sampleName,batchSize,true,false) 
61         tests += testGen(sampleName,batchSize,true,true) 
62       } 
63       println(s"Query - group(lineitem). Dataset - $sampleName") 
64       tests.foreach(singleDStreamTest) 
65       println() 
66     }

177   
178      /**Q1 with ONLY lineitem streamed*/ 
179      def genQuery[V,ID](rdds: TpchRdds, lineitem: LiteralExpr[V,ID]) = { 
180        val customer = Bag(rdds.customer, "customer".narrow) 
181        val orders = Bag(rdds.orders, "orders".narrow) 
182        val part = Bag(rdds.part, "part".narrow) 
183   
184        val (orderKey0, partName0) = Vars("X1", "X2") 
185        val (partNames0, custKey0, orderDate0) = Vars("Y1", "Y2", "Y3") 
186        val (orderDate1, partNames1, custName0) = Vars("Z1", "Z2", "Z3") 
187   
188        //Drop the suppkeys to get a Bag[(partkey,orderkey)] 
189        val partOrders = For((l_orderkey, l_partkey, __) <-- lineitem) Yield(l_partkey, l_orderkey) 
190   
191        //join with part to get a Bag[(partkey,(orderKey,partName))], drop the partkeys and group to get 
192        //a Bag[(orderKey,Bag[partName])] 
193        val orderPartNames = Group( 
194          For((__, orderKey0, partName0) <-- partOrders.join(part)) Yield(orderKey0, partName0) 
195        ) 
196        //join with orders and reshape to get a Bag[(custKey,orderDate,Bag[partName])] 
197        val customerOrders = 
198          For((__, partNames0, custKey0, orderDate0) <-- orderPartNames.join(orders)) Yield(custKey0, orderDate0, partNames0) 
199   
200        //join with customers and reshape to get a Bag[(custName,orderDate,Bag[partName])] 
201        val customerNameOrders = 
202          For((__, orderDate1, partNames1, custName0, __) <-- customerOrders.join(customer)) Yield 
203            (custName0, orderDate1, partNames1) 
204   
205        //finally group to get a Bag[(custName,Bag[(orderDate,Bag[partName])])] 
206        val query = Group(customerNameOrders) 
207        query 
208      }

238      /**Q2 with ONLY orders streamed*/ 
239      def genQuery[V,ID](rdds: TpchRdds, orders: LiteralExpr[V,ID]) = { 
240        val custNames = Var("X") 
241        val lineitem = Bag(rdds.lineitem, "lineitem".narrow) 
242        val customer = Bag(rdds.customer, "customer".narrow) 
243        val supplier = Bag(rdds.supplier, "supplier".narrow) 
244   
245        //First join orders with lineitems on orderkey to get all pairs of custkey/suppkey 
246        val custSuppliers = For ((__,o_custkey,__,__,l_suppkey) <-- orders.join(lineitem)) Yield (o_custkey,l_suppkey) 
247        //Then join customer with the above to get the customers names, yielding all pairs of suppkey/custname. 
248        //Then group this to get each supplier with its bag of customer names who have used it: 
249        val inner = For ((__,c_name,__,l_suppkey) <-- customer.join(custSuppliers)) Yield (l_suppkey,c_name) 
250        val supplierCustomers = Group( 
251          For ((__,c_name,__,l_suppkey) <-- customer.join(custSuppliers)) Yield (l_suppkey,c_name) 
252        ) 
253        //Finally join with supplier to get the suppliers names: 
254        val query = For ((__,s_name,__,custNames) <-- supplier.join(supplierCustomers)) Yield (s_name,custNames) 
255        query 
256      }

287      def genQuery[V,ID](rdds: TpchRdds, partSupp: LiteralExpr[V,ID]) = { 
288        /**For each part, return the name, the names and nations of all suppliers who supply it, 
289        * and the names and nations of all customers who've ordered it 
290        * */ 
291   
292        val (suppliers,customers) = Vars("SUPS","CUSTS") 
293        val lineitem = Bag(rdds.lineitem, "lineitem".narrow) 
294        val customer = Bag(rdds.customer, "customer".narrow) 
295        val supplier = Bag(rdds.supplier, "supplier".narrow) 
296        val part = Bag(rdds.part, "part".narrow) 
297        val orders = Bag(rdds.orders, "orders".narrow) 
298   
299        //First join partSupp to suppliers and group to get partkeys and their bags of supplier names 
300        val partSupp2 = For ((ps_partkey,ps_suppkey) <-- partSupp) Yield (ps_suppkey,ps_partkey) 
301        val partSuppliers = Group( 
302          For ((ps_suppkey,ps_partkey,s_name,s_nationkey) <-- partSupp2.join(supplier)) Yield (ps_partkey,(s_name,s_nationkey)) 
303        ) 
304   
305        val partCustomerPairs = For ((__,l_partkey,__,o_custkey,__) <-- lineitem.join(orders)) Yield (l_partkey,o_custkey) 
306        val partCustomers = Group( 
307          For ((l_partkey,__,c_name,c_nationkey) <-- partCustomerPairs.join(customer)) Yield (l_partkey,(c_name,c_nationkey)) 
308        ) 
309   
310        val query = 
311          For ( 
312            (__,p_name,suppliers,customers) <-- part.join(partSuppliers.join(partCustomers)) 
313          ) Yield (p_name,suppliers,customers) 
314   
315        query 
316      }
